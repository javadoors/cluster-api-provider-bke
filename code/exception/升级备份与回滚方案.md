# 升级备份与回滚方案

## 一、设计目标

1. **自动备份**：升级前自动备份集群配置、版本信息和关键状态
2. **快速回滚**：升级失败时能快速回滚到升级前状态
3. **最小侵入**：与现有 DAG 升级流程无缝集成
4. **可观测性**：提供备份历史和回滚操作的完整审计日志

## 二、备份机制设计

### 2.1 备份内容

| 备份项 | 存储位置 | 说明 |
|--------|---------|------|
| **集群配置快照** | `BKECluster.Status.PreUpgradeSnapshot.Config` | BKECluster.Spec 的完整 JSON 快照 |
| **版本信息** | `BKECluster.Status.PreUpgradeSnapshot.Versions` | 各组件当前版本（K8s、etcd、Containerd、OpenFuyao） |
| **升级目标** | `BKECluster.Status.PreUpgradeSnapshot.TargetVersion` | 本次升级的目标版本 |
| **备份时间** | `BKECluster.Status.PreUpgradeSnapshot.Timestamp` | 备份创建时间 |
| **备份 ID** | `BKECluster.Status.PreUpgradeSnapshot.BackupID` | 唯一标识符（UUID） |

```go
// createPreUpgradeBackup 创建升级前备份
func (r *BKEClusterReconciler) createPreUpgradeBackup(
    ctx context.Context,
    cluster *bkev1beta1.BKECluster,
) error {
    // 生成备份 ID
    backupID := uuid.New().String()
    
    // 序列化集群配置
    configJSON, err := json.Marshal(cluster.Spec)
    if err != nil {
        return errors.Wrap(err, "failed to marshal cluster spec")
    }
    
    // 创建备份快照
    snapshot := &bkev1beta1.PreUpgradeSnapshot{
        BackupID:      backupID,
        Timestamp:     metav1.Now(),
        TargetVersion: cluster.Annotations["cvo.openfuyao.cn/upgrade-ready"],
        Config:        string(configJSON),
        Versions: bkev1beta1.ComponentVersions{
            KubernetesVersion: cluster.Status.KubernetesVersion,
            EtcdVersion:       cluster.Status.EtcdVersion,
            ContainerdVersion: cluster.Status.ContainerdVersion,
            OpenFuyaoVersion:  cluster.Status.OpenFuyaoVersion,
        },
        PhaseStatus:        cluster.Status.PhaseStatus.DeepCopy(),
        DeclarativeUpgrade: cluster.Status.DeclarativeUpgrade.DeepCopy(),
    }
    
    // 保存到 Status
    cluster.Status.PreUpgradeSnapshot = snapshot
    
    // 记录事件
    r.Recorder.Eventf(cluster, corev1.EventTypeNormal, "PreUpgradeBackupCreated",
        "Created pre-upgrade backup with ID: %s", backupID)
    
    return nil
}
```

## 三、回滚机制设计

### 3.1 回滚触发方式

| 触发方式 | 说明 | 实现 |
|---------|------|------|
| **手动回滚** | 用户通过注解触发 | `bke.bocloud.com/rollback=<backupID>` |
| **自动回滚** | 升级失败时自动触发 | 在 DAG 执行失败后自动回滚 |
| **定时回滚** | 升级后健康检查失败 | 升级完成后 N 分钟内健康检查失败则回滚 |

### 3.2 回滚流程

```
用户设置 rollback 注解
    |
    v
BKEClusterReconciler 检测到注解
    |
    v
验证备份是否存在
    |
    v
创建回滚记录（RollbackRecord）
    |
    v
恢复集群配置（Spec）
    |
    v
恢复版本信息（Status）
    |
    v
重置升级状态
    |
    v
清除 upgrade-ready 注解
    |
    v
更新回滚记录状态
    |
    v
记录回滚事件
```

### 3.3 回滚实现

**文件：`controllers/capbke/bkecluster_rollback.go`（新文件）**

```go
package capbke

import (
    "context"
    "encoding/json"
    
    "github.com/pkg/errors"
    corev1 "k8s.io/api/core/v1"
    metav1 "k8s.io/apimachinery/pkg/apis/meta/v1"
    ctrl "sigs.k8s.io/controller-runtime"
    
    bkev1beta1 "gopkg.openfuyao.cn/cluster-api-provider-bke/api/capbke/v1beta1"
    "gopkg.openfuyao.cn/cluster-api-provider-bke/utils/capbke/annotation"
    "gopkg.openfuyao.cn/cluster-api-provider-bke/utils/log"
)

const (
    // RollbackAnnotationKey 回滚注解键
    RollbackAnnotationKey = "bke.bocloud.com/rollback"
    
    // RollbackStatusInProgress 回滚进行中
    RollbackStatusInProgress = "InProgress"
    // RollbackStatusCompleted 回滚完成
    RollbackStatusCompleted = "Completed"
    // RollbackStatusFailed 回滚失败
    RollbackStatusFailed = "Failed"
)

// executeRollback 执行回滚
func (r *BKEClusterReconciler) executeRollback(
    ctx context.Context,
    cluster *bkev1beta1.BKECluster,
    backupID string,
) (ctrl.Result, error) {
    logger := log.With("cluster", cluster.Name, "backupID", backupID)
    
    // 1. 验证备份是否存在
    if cluster.Status.PreUpgradeSnapshot == nil {
        return ctrl.Result{}, errors.New("no pre-upgrade backup found")
    }
    
    if cluster.Status.PreUpgradeSnapshot.BackupID != backupID {
        return ctrl.Result{}, errors.Errorf("backup ID mismatch: expected %s, got %s",
            cluster.Status.PreUpgradeSnapshot.BackupID, backupID)
    }
    
    // 2. 创建回滚记录
    rollbackRecord := &bkev1beta1.RollbackRecord{
        RollbackID:  uuid.New().String(),
        Timestamp:   metav1.Now(),
        FromVersion: cluster.Status.OpenFuyaoVersion,
        ToVersion:   cluster.Status.PreUpgradeSnapshot.Versions.OpenFuyaoVersion,
        Reason:      "Manual rollback triggered by user",
        Status:      RollbackStatusInProgress,
    }
    
    // 3. 恢复集群配置
    if err := r.restoreClusterConfig(ctx, cluster); err != nil {
        rollbackRecord.Status = RollbackStatusFailed
        rollbackRecord.ErrorMessage = err.Error()
        cluster.Status.RollbackHistory = append(cluster.Status.RollbackHistory, *rollbackRecord)
        return ctrl.Result{}, errors.Wrap(err, "failed to restore cluster config")
    }
    
    // 4. 恢复版本信息
    r.restoreVersionInfo(cluster)
    
    // 5. 重置升级状态
    r.resetUpgradeStatus(cluster)
    
    // 6. 清除升级注解
    delete(cluster.Annotations, "cvo.openfuyao.cn/upgrade-ready")
    delete(cluster.Annotations, RollbackAnnotationKey)
    
    // 7. 更新回滚记录
    rollbackRecord.Status = RollbackStatusCompleted
    cluster.Status.RollbackHistory = append(cluster.Status.RollbackHistory, *rollbackRecord)
    
    // 8. 记录事件
    r.Recorder.Eventf(cluster, corev1.EventTypeNormal, "RollbackCompleted",
        "Rollback completed successfully from %s to %s",
        rollbackRecord.FromVersion, rollbackRecord.ToVersion)
    
    logger.Info("Rollback completed successfully")
    
    return ctrl.Result{}, nil
}

// restoreClusterConfig 恢复集群配置
func (r *BKEClusterReconciler) restoreClusterConfig(
    ctx context.Context,
    cluster *bkev1beta1.BKECluster,
) error {
    snapshot := cluster.Status.PreUpgradeSnapshot
    
    // 反序列化配置
    var spec bkev1beta1.BKEClusterSpec
    if err := json.Unmarshal([]byte(snapshot.Config), &spec); err != nil {
        return errors.Wrap(err, "failed to unmarshal cluster spec")
    }
    
    // 恢复配置
    cluster.Spec = spec
    
    return nil
}

// restoreVersionInfo 恢复版本信息
func (r *BKEClusterReconciler) restoreVersionInfo(cluster *bkev1beta1.BKECluster) {
    snapshot := cluster.Status.PreUpgradeSnapshot
    
    cluster.Status.KubernetesVersion = snapshot.Versions.KubernetesVersion
    cluster.Status.EtcdVersion = snapshot.Versions.EtcdVersion
    cluster.Status.ContainerdVersion = snapshot.Versions.ContainerdVersion
    cluster.Status.OpenFuyaoVersion = snapshot.Versions.OpenFuyaoVersion
}

// resetUpgradeStatus 重置升级状态
func (r *BKEClusterReconciler) resetUpgradeStatus(cluster *bkev1beta1.BKECluster) {
    // 重置集群状态
    cluster.Status.ClusterStatus = bkev1beta1.ClusterReady
    cluster.Status.ClusterHealthState = bkev1beta1.Healthy
    
    // 重置阶段状态
    if cluster.Status.PreUpgradeSnapshot.PhaseStatus != nil {
        cluster.Status.PhaseStatus = cluster.Status.PreUpgradeSnapshot.PhaseStatus
    }
    
    // 重置声明式升级状态
    if cluster.Status.PreUpgradeSnapshot.DeclarativeUpgrade != nil {
        cluster.Status.DeclarativeUpgrade = cluster.Status.PreUpgradeSnapshot.DeclarativeUpgrade
    }
    
    // 清除备份快照（可选：保留用于审计）
    // cluster.Status.PreUpgradeSnapshot = nil
}
```

```go
    ctx context.Context,
    phaseCtx *phaseframe.PhaseContext,
    oldCluster, newCluster *bkev1beta1.BKECluster,
) (ctrl.Result, error) {
    // 创建备份
    if err := r.createPreUpgradeBackup(ctx, newCluster); err != nil {
        return ctrl.Result{}, errors.Wrap(err, "failed to create pre-upgrade backup")
    }
    
    // ... 执行升级 ...
    
    if err := sched.ExecuteDAG(ctx, phaseCtx, oldCluster, newCluster, dag); err != nil {
        // 标记升级失败
        _ = r.patchClusterStatus(newCluster, bkev1beta1.ClusterUpgradeFailed)
        
        // 检查是否需要自动回滚
        if r.shouldAutoRollback(newCluster) {
            logger.Info("Auto rollback triggered due to upgrade failure")
            return r.executeRollback(ctx, newCluster, newCluster.Status.PreUpgradeSnapshot.BackupID)
        }
        
        if res, requeue := dagexec.RequeueAwareError(err); requeue {
            return res, err
        }
        return ctrl.Result{}, err
    }
    
    // ... 完成升级 ...
}

        return true
```

### 4.2 升级后健康检查回滚

**文件：`controllers/capbke/bkecluster_post_upgrade.go`（新文件）**

```go
package capbke

import (
    "context"
    "time"
    
    bkev1beta1 "gopkg.openfuyao.cn/cluster-api-provider-bke/api/capbke/v1beta1"
)

const (
    // PostUpgradeHealthCheckWindow 升级后健康检查窗口
    PostUpgradeHealthCheckWindow = 10 * time.Minute
)

// checkPostUpgradeHealth 检查升级后健康状态
func (r *BKEClusterReconciler) checkPostUpgradeHealth(
    ctx context.Context,
    cluster *bkev1beta1.BKECluster,
) (bool, error) {
    // 检查是否在健康检查窗口内
    if cluster.Status.DeclarativeUpgrade == nil {
        return true, nil
    }
    
    finishedAt := cluster.Status.DeclarativeUpgrade.FinishedAt
    if finishedAt == nil {
        return true, nil
    }
    
    elapsed := time.Since(finishedAt.Time)
    if elapsed > PostUpgradeHealthCheckWindow {
        // 超过窗口期，不再检查
        return true, nil
    }
    
    // 执行健康检查
    healthy, err := r.performHealthCheck(ctx, cluster)
    if err != nil {
        return false, err
    }
    
    if !healthy {
        // 健康检查失败，触发回滚
        return false, nil
    }
    
    return true, nil
}
```

## 八、风险评估

| 风险 | 概率 | 影响 | 缓解措施 |
|------|------|------|---------|
| 备份数据过大 | 中 | 中 | 只备份必要字段，定期清理旧备份 |
| 回滚失败 | 低 | 高 | 回滚前验证备份完整性，失败时保留现场 |
| 配置冲突 | 中 | 中 | 回滚前检查是否有外部配置变更 |
| 数据丢失 | 低 | 高 | 备份数据持久化到 etcd，定期备份到外部存储 |

## 十、总结

本方案提供了完整的备份和回滚机制，与现有 DAG 升级流程无缝集成：

1. **备份机制**：升级前自动备份集群配置、版本信息和状态
2. **手动回滚**：用户通过注解触发回滚
3. **自动回滚**：升级失败或健康检查失败时自动回滚
4. **审计日志**：记录所有回滚操作的历史

通过本方案，可以显著提高升级的安全性和可恢复性，降低升级失败带来的风险。
